/**
 * 获取分类视频列表
 * @param {UZArgs} args
 * @returns {@Promise<JSON.stringify(new RepVideoList())>}
 */
async function getVideoList(args) {
    var backData = new RepVideoList()
    backData.data = []
    
    try {
        let webSite = UZUtils.removeTrailingSlash(appConfig.webSite)
        let page = args.page || 1
        let url = `${webSite}/video/query?page=${page}&limit=20&tid=${args.url}&class=&year=&order=time`
        
        UZUtils.debugLog('请求URL: ' + url)
        
        let response = await req(url, {
            method: 'GET',
            headers: appConfig.headers
        })
        
    
        
        // 关键修复：处理 response.data 可能已经是对象的情况
        let json = response.data
      
        
        UZUtils.debugLog('返回数据条数: ' + (json.data ? json.data.length : 0))
        
        if (json.data && Array.isArray(json.data)) {
            let videos = []
            
            for (let i = 0; i < json.data.length; i++) {
                let item = json.data[i]
                
                // 关键修复：确保所有必需字段都存在且类型正确
                let video = {
                    vod_id: String(item.vod_id),
                    vod_name: String(item.name || ''),
                    vod_pic: String(item.pic || ''),
                    vod_remarks: String(item.score || item.remarks || ''),
                    // 可选字段
                    type_name: '',
                    vod_actor: '',
                    vod_director: '',
                    vod_area: '',
                    vod_content: '',
                    vod_year: '',
                    vod_play_from: '',
                    vod_play_url: ''
                }
                
                videos.push(video)
            }
            
            // 关键修复：直接赋值数组，不要包装
            backData.data = videos
            backData.total = json.total || videos.length
            UZUtils.debugLog('解析成功，共 ' + videos.length + ' 条')
        } else {
            backData.error = '返回数据格式错误'
        }
        
    } catch (error) {
        backData.error = '获取列表失败: ' + error.toString()
        UZUtils.debugLog(backData.error)
    }
    
    return JSON.stringify(backData)
}

/**
 * 获取视频详情
 * @param {UZArgs} args
 * @returns {@Promise<JSON.stringify(new RepVideoDetail())>}
 */
async function getVideoDetail(args) {
    var backData = new RepVideoDetail()
    
    try {
        let webSite = UZUtils.removeTrailingSlash(appConfig.webSite)
        let detailUrl = `${webSite}/video/info/${args.url}`
        
        UZUtils.debugLog('详情URL: ' + detailUrl)
        
        let response = await req(detailUrl, {
            method: 'GET',
            headers: appConfig.headers
        })
        
   
        
        // 关键修复：处理 response.data 可能已经是对象的情况
        let data = response.data
     
        
        let items = data.data
        
        if (!items) {
            backData.error = '未获取到详情数据'
            return JSON.stringify(backData)
        }
        
        let vod_play_from = []
        let vod_play_url = []
        
        // 处理播放列表
        if (items.vod_play_from && Array.isArray(items.vod_play_from)) {
            for (let i = 0; i < items.vod_play_from.length; i++) {
                let from = items.vod_play_from[i]
                let fromName = from.name || from.code || '线路' + (i + 1)
                vod_play_from.push(fromName)
                
                try {
                    let playRes = await req(`${webSite}/video/play_url?id=${args.url}&from=${from.code}`, {
                        method: 'GET',
                        headers: appConfig.headers
                    })
                    
                   
                    
                    // 关键修复：处理 playRes.data 可能已经是对象的情况
                    let playData = playRes.data
                 
               
                    if (playData.data && Array.isArray(playData.data)) {
                        let playlist = playData.data.map(item => {
                            return `${item.name}$${item.url}`
                        }).join('#')
                        vod_play_url.push(playlist)
                    }
                    UZUtils.debugLog(vod_play_url)
                } catch (e) {
                    UZUtils.debugLog('获取播放地址失败: ' + e.message)
                }
            }
        }
        
        // 构建详情对象
        let video = {
            vod_id: String(args.url),
            vod_name: String(items.vod_name || ''),
            vod_pic: String(items.vod_pic || ''),
            type_name: String(items.vod_class || ''),
            vod_actor: String(items.vod_actor || ''),
            vod_director: String(items.vod_director || ''),
            vod_area: String(items.vod_area || ''),
            vod_content: String(items.vod_blurb || ''),
            vod_remarks: String(items.vod_remarks || ''),
            vod_year: String(items.vod_year || ''),
            vod_play_from: vod_play_from.join('$$$'),
            vod_play_url: vod_play_url.join('$$$')
        }
        
        backData.data = video
        
    } catch (error) {
        backData.error = '获取详情失败: ' + error.toString()
        UZUtils.debugLog(backData.error)
    }
    
    return JSON.stringify(backData)
}

/**
 * 搜索视频
 * @param {UZArgs} args
 * @returns {@Promise<JSON.stringify(new RepVideoList())>}
 */
async function searchVideo(args) {
    var backData = new RepVideoList()
    backData.data = []
    
    try {
        let webSite = UZUtils.removeTrailingSlash(appConfig.webSite)
        let url = `${webSite}/video/search?text=${encodeURIComponent(args.searchWord)}&pg=${args.page || 1}&limit=30`
        
        UZUtils.debugLog('搜索URL: ' + url)
        
        let response = await req(url, {
            method: 'GET',
            headers: appConfig.headers
        })
        
        
        
        // 关键修复：处理 response.data 可能已经是对象的情况
        let data = response.data
     
        
        if (data.data && Array.isArray(data.data)) {
            let videos = data.data.map(item => {
                return {
                    vod_id: String(item.vod_id),
                    vod_name: String(item.name || ''),
                    vod_pic: String(item.pic || ''),
                    vod_remarks: String(item.year || ''),
                    type_name: '',
                    vod_actor: '',
                    vod_director: '',
                    vod_area: '',
                    vod_content: '',
                    vod_year: '',
                    vod_play_from: '',
                    vod_play_url: ''
                }
            })
            backData.data = videos
            backData.total = data.total || videos.length
        }
        
    } catch (error) {
        backData.error = '搜索失败: ' + error.toString()
        UZUtils.debugLog(backData.error)
    }
    
    return JSON.stringify(backData)
}
